@Directive({
    selector: '[github-avatar]',
    providers: [ElementRef]
})
export class AvatarComponent implements OnInit {
    private img: HTMLImageElement;
 
    constructor(private _el: ElementRef) {}
 
    @Input('github-avatar')
    set src(src: String) {
        this.img = new Image();
        this.img.crossOrigin = 'anonymous';
        this.img.onload = () => {
            var data = this._getImgBase64(this.img);
            this.setHostSrcData(data);
            this._el.nativeElement.style.opacity = 1;
        };
        this.img.src = <string>src;
    }
 
    setHostSrcData(data: String) {
        this._el.nativeElement.src = data;
    }
 
    ngOnInit() {
        this._el.nativeElement.style.opacity = 0.3;
        this.setHostSrcData(octoCat);
    }
 
    private _getImgBase64(img: HTMLImageElement): String {
        var width = this.img.width;
        var height = this.img.height;
        var canvas = <HTMLCanvasElement>document.createElement('CANVAS');
        canvas.width = width;
        canvas.height = height;
        var ctx = canvas.getContext('2d');
        ctx.drawImage(img, 0, 0, width, height);
        return canvas.toDataURL('image/png')
    }
}
